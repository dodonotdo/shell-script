```bash
if Statement:
-------------
a=10
b=20
  
if [ $a == $b ]
then
    echo "a is equal to b"
fi
  
if [ $a != $b ]
then
    echo "a is not equal to b"
fi

output: 
    a is not equal to b

```

```bash
if else statement:
------------------
#Initializing two variables
a=20
b=20
  
if [ $a == $b ]
then
    #If they are equal then print this
    echo "a is equal to b"
else
    #else print this
    echo "a is not equal to b"
fi

output:
    a is equal to b
```
